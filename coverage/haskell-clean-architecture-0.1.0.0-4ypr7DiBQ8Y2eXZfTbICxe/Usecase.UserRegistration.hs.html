   12 -- PUBLIC
   13 -- the pure logic usecase signature
   14 type Register m = Monad m => Text -> Text -> Text -> m (Either [D.Error] Text)
   15 
   16 -- the usecase
   17 register
   18   :: (UC.Logger m, MonadThrow m, MonadUnliftIO m, Exception Err)
   19   => UC.GenUUID m
   20   -> UC.CheckEmailFormat m
   21   -> UC.GetUserByEmail m
   22   -> UC.GetUserByName m
   23   -> UC.InsertUserPswd m
   24   -> Register m
   25 register genUUID checkEmail getUserByEmail getUserByName insertUserPswd name email pswd = catch
   26 
   27   (do
   28     malformedEmailRes <- checkEmail email
   29     collidingEmailRes <- checkNoCollidingEmail getUserByEmail email
   30     collidingNameRes  <- checkNoCollidingName getUserByName name
   31   -- we want to accumulate the potential errors of the previous steps
   32     checkValidation [malformedEmailRes, collidingEmailRes, collidingNameRes]
   33     uuid <- genUUID
   34     insertUser insertUserPswd (D.User uuid name email) pswd
   35     pure $ Right uuid
   36   )
   37 
   38   (\e -> do
   39     err <- handleExceptions e
   40     pure $ Left err
   41   )
   42 
   43 -- PRIVATE
   44 data Err = ErrValidation [D.Error]
   45   | ErrTechnical
   46     deriving (Show, Eq)
   47 instance Exception Err
   48 
   49 handleExceptions :: (UC.Logger m, MonadUnliftIO m, Exception Err) => Err -> m [D.Error]
   50 handleExceptions e = case e of
   51   ErrTechnical -> do
   52     UC.log [e]
   53     pure [D.ErrTechnical]
   54   ErrValidation errs -> do
   55     UC.log errs
   56     pure errs
   57 
   58 checkNoCollidingEmail
   59   :: (MonadThrow m, Exception Err) => UC.GetUserByEmail m -> Text -> m (Maybe [D.Error])
   60 checkNoCollidingEmail getUserByEmail email = do
   61   mayUser <- getUserByEmail email
   62   case mayUser of
   63     Left  _        -> throwM ErrTechnical
   64     Right Nothing  -> pure Nothing
   65     Right (Just _) -> pure $ Just [D.ErrEmailConflict]
   66 
   67 checkNoCollidingName
   68   :: (MonadThrow m, Exception Err) => UC.GetUserByName m -> Text -> m (Maybe [D.Error])
   69 checkNoCollidingName getUserByName name = do
   70   mayUser <- getUserByName name
   71   case mayUser of
   72     Left  _        -> throwM ErrTechnical
   73     Right Nothing  -> pure Nothing
   74     Right (Just _) -> pure $ Just [D.ErrNameConflict]
   75 
   76 checkValidation :: (MonadThrow m, Exception Err) => [Maybe [D.Error]] -> m ()
   77 checkValidation errs = case combine errs of
   78   [] -> pure ()
   79   e  -> throwM (ErrValidation e)
   80 
   81 combine :: [Maybe [a]] -> [a]
   82 combine = concatMap concat
   83 
   84 insertUser :: (MonadThrow m, Exception Err) => UC.InsertUserPswd m -> D.User -> Text -> m ()
   85 insertUser insert user pswd = do
   86   res <- insert user pswd
   87   case res of
   88     Nothing -> pure ()
   89     Just _  -> throwM ErrTechnical

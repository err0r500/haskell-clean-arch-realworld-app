   12 class Show a => Loggable a where
   13   type F a
   14   log' :: a -> F (IO ())
   15   show' :: a -> Text
   16 
   17 instance Show a => Loggable a where
   18   type F a = KatipContextT IO ()
   19   log' mess = $(logTM) ErrorS (showLS mess)
   20   show' = tshow
   21 
   22 instance Loggable D.Error where
   23   type F D.Error = KatipContextT IO ()
   24   log' err = $(logTM) ErrorS (showLS err)
   25   show' = tshow
   26 
   27 instance Show a => Loggable ( D.Message a ) where
   28   type F (D.Message a) = KatipContextT IO ()
   29   log' (D.ErrorMsg   mess) = $(logTM) ErrorS (showLS $ show mess)
   30   log' (D.WarningMsg mess) = $(logTM) WarningS (showLS $ show mess)
   31   log' (D.InfoMsg    mess) = $(logTM) InfoS (showLS $ show mess)
   32   log' (D.DebugMsg   mess) = $(logTM) DebugS (showLS $ show mess)
   33   show' = tshow
   34 
   35 log :: (MonadIO m, Loggable a) => [a] -> m ()
   36 log elemsToLog = do
   37   handleScribe <- liftIO $ mkHandleScribe ColorIfTerminal stdout (permitItem DebugS) V2
   38   let mkLogEnv = registerScribe "stdout" handleScribe defaultScribeSettings
   39         =<< initLogEnv "MyApp" "production"
   40   liftIO $ Control.Exception.bracket mkLogEnv closeScribes $ \le ->
   41     runKatipContextT le () mempty $ mapM_ log' elemsToLog

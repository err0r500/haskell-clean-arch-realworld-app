   17 start :: (MonadIO m, UC.Logger m, MonadThrow m ) => Lib.Router m
   18 start logicHandler runner =
   19   pure $
   20   serve (Proxy :: Proxy API) $
   21   hoistServer (Proxy :: Proxy API) ( ioToHandler . runner ) (server logicHandler)
   22 
   23 
   24 type API =
   25   Get '[JSON] NoContent
   26   :<|> Capture "name" Text :> Capture "email" Text :> Capture "password" Text :> Post '[PlainText] Text
   27 
   28 
   29 server :: (MonadThrow m, MonadUnliftIO m, MonadIO m, UC.Logger m) => UC.LogicHandler m -> ServerT API m
   30 server  logicHandler = healthH :<|> regUser
   31 
   32   where
   33     healthH = pure NoContent
   34     regUser = Router.registerUser (UC._userRegister logicHandler)
   35

   37 -- liftIO is not enough to catch the "throwM"s in handlers so we rebuild the handler manually
   38 ioToHandler :: IO a -> Handler a
   39 ioToHandler = Handler . Except.ExceptT . try

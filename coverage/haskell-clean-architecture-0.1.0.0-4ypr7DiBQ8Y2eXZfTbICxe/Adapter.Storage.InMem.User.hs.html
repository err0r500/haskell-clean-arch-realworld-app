   68 commonSearch :: InMemory r m => (User -> Bool) -> m (Either D.Error (Maybe D.User))
   69 commonSearch filter_ = do
   70   tvar <- asks DH.getter
   71   atomically $ do
   72     state <- readTVar tvar
   73     case filter filter_ $ map snd $ Map.toList (users state) of
   74       []      -> pure $ Right Nothing
   75       (x : _) -> pure $ Right (Just (toDomain x))
